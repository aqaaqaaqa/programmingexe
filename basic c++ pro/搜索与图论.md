###dfs代码

> dfs在回溯时一定要恢复现场

```c++
void dfs(int u){
    if(u == n){
        for(int i = 0; i < n; i++) printf("%d", path[i]);
        puts(" ");
        return;
    }
    
	for(int i = 1; i < n; i++){
		if(!st[i]){
			path[u] = i;
			str[i] = true;
			dfs(u + 1);
			st[i] = false;
		}
	}
}
```
